<template>
  <div class="product-display" @change="">
    <div class="products-container">
      <div v-for="(product, index) in products" :key="product.id" class="product-item">
        <el-card v-if="!product.loaded" class="cardclass" shadow="hover">
          <template #header>
            <el-skeleton-item variant="text" style="width: 30%;"/>
          </template>
          <div class="skeleton-content">
            <div class="skeleton-image">
              <el-skeleton-item variant="image" style="width: 100%; height: 100%;"/>
            </div>
            <div class="skeleton-right">
              <div class="skeleton-info">
                <el-skeleton-item variant="text" style="width: 100%;"/>
                <el-skeleton-item variant="text" style="width: 80%; margin-top: 10px"/>
              </div>
              <div class="skeleton-buttons">
                <el-skeleton-item variant="button" style="width: 80px;"/>
                <el-skeleton-item variant="button" style="width: 80px;"/>
              </div>
            </div>
          </div>
        </el-card>
        <el-card v-else class="cardclass" shadow="hover">
          <template #header>
            <span v-if="category==='payment'">{{product.name}}——剩余时间{{product.time}}</span>
            <span v-else>{{product.name}}</span>
          </template>
          <div class="card-content">
            <div class="product-image">
              <img :src="product.imageUrl"
                   @load="onImageLoad($event, product)"
                   @error="onImageError(product)"/>
            </div>
            <div class="product-right">
              <div class="product-info">卖家：牛魔———价格：{{product.price}}元</div>
              <div class="price-button-container">
                <el-button v-if="category==='payment'" type="warning" plain>付款</el-button>
                <el-button v-else-if="category==='delivering'" type="warning" plain>退款</el-button>
                <el-button v-else type="warning" plain>退货</el-button>
                <el-button type="danger" plain>取消</el-button>
              </div>
            </div>
          </div>
        </el-card>
      </div>
    </div>
  </div>
</template>

<script lang="ts" setup>
import { ref, onMounted, defineEmits, defineProps, watch, reactive } from 'vue';
import { useCounterStore } from '@/stores/counter';  // 引入 Pinia store
import type { CheckboxValueType } from 'element-plus'

const emit = defineEmits(['product-change']);
const props = defineProps({
  category: String,
});

// 获取 Pinia store
const productsStore = useCounterStore();

const products = reactive([])

onMounted(() => {
  if(props.category=="payment"){
    products.value = productsStore.payproducts;
  }else if(props.category==="delivering"){
    products.value = productsStore.deliveringproducts;
  }else{
    products.value = productsStore.deliveredproducts;
  }
});

// 图片加载成功处理函数
const onImageLoad = (event, product) => {
  const img = event.target;
  if (img.naturalWidth > 0 && img.naturalHeight > 0) {
    console.log(`图片 ${product.imageUrl} 加载清晰`);
    product.loaded = true;
  } else {
    console.log(`图片 ${product.imageUrl} 可能未完全加载`);
  }
  console.log("加载出来");
};

// 图片加载失败处理函数
const onImageError = (product) => {
  console.error(`图片加载失败: ${product.imageUrl}`);
  product.imageUrl = 'https://via.placeholder.com/200x200';
  product.loaded = true;
};

onMounted(() => {
  products.forEach((product) => {
    const img = new Image();
    img.src = product.imageUrl;
    console.log("挂载"+product.imageUrl);

    // 检查图片是否已经加载完成
    if (img.complete) {
      onImageLoad({ target: img }, product);
    } else {
      // 图片未加载完成，绑定 onload 和 onerror 事件
      img.onload = () => {
        onImageLoad({ target: img }, product);
      };
      img.onerror = () => {
        onImageError(product);
      };
    }
  });
});
</script>
